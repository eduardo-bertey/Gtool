REM --- Comandos de una linea para compilar cada arquitectura de Android ---
REM --- Ejecutar desde el directorio 'rust' ---

REM --- aarch64-linux-android (ARM 64-bit) ---
REM DEBUG
set CARGO_TARGET_AARCH64_LINUX_ANDROID_LINKER=%ANDROID_NDK_HOME%\toolchains\llvm\prebuilt\windows-x86_64\bin\aarch64-linux-android34-clang.cmd && set CLANG_PATH=%ANDROID_NDK_HOME%\toolchains\llvm\prebuilt\windows-x86_64\bin\clang.cmd && cargo build --target=aarch64-linux-android
REM RELEASE
set CARGO_TARGET_AARCH64_LINUX_ANDROID_LINKER=%ANDROID_NDK_HOME%\toolchains\llvm\prebuilt\windows-x86_64\bin\aarch64-linux-android34-clang.cmd && set CLANG_PATH=%ANDROID_NDK_HOME%\toolchains\llvm\prebuilt\windows-x86_64\bin\clang.cmd && cargo build --release --target=aarch64-linux-android
